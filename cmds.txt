# Prepare CMD and Powershell for development:
cmd > "C:\Program Files\Microsoft Visual Studio\2022\Community\VC\Auxiliary\Build\vcvarsall.bat" amd64 10.0.19041.0
cmd > pwsh.exe    # Starts PS by previous setting up the env vars.
ps  > gci env:    # Retrives Environment Info.

# Remove build content
gci .\build\ | ForEach-Object FullName | Remove-Item

# Precompile pch.cpp

& cl.exe /c /ZI /diagnostics:column /std:c++17 /W4 /Od /D _DEBUG /D _WINDOWS /D _UNICODE /D UNICODE /Gm- /EHsc /fp:precise /MDd /external:W4 /FC /errorReport:prompt /Yc"pch.h" /Fp"build\pch.pch" /Fd"build\" /Fo"build\pch.obj"  ".\pch.cpp"

# Compile main.cpp

& cl.exe /c /ZI /diagnostics:column /std:c++17 /W4 /Od /D _DEBUG /D _WINDOWS /D _UNICODE /D UNICODE /Gm- /EHsc /fp:precise /MDd /external:W4 /FC /errorReport:prompt /Yu"pch.h" /Fp"build\pch.pch" /Fo"build\" /Fd"build\" ".\main.cpp"

# Link to main.exe

& link.exe /ERRORREPORT:PROMPT /OUT:"build\main.exe" /INCREMENTAL /ILK:"build\main.ilk" dwmapi.lib kernel32.lib user32.lib gdi32.lib winspool.lib comdlg32.lib advapi32.lib shell32.lib ole32.lib oleaut32.lib uuid.lib odbc32.lib odbccp32.lib /MANIFEST /MANIFESTUAC:"level='asInvoker' uiAccess='false'" /manifest:embed /DEBUG /PDB:"build\vc140.pdb" /SUBSYSTEM:CONSOLE /TLBID:1 /DYNAMICBASE /NXCOMPAT /IMPLIB:"build\main.lib" /MACHINE:X64 "build\main.obj" "build\pch.obj"
